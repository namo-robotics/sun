## Debug Mode

Use `--debug` to generate debugging artifacts in a `<filename>_debug/` folder:

```bash
sun --debug hello.sun
```

This creates:
- `ast.dot` — GraphViz visualization of the AST
- `ir.ll` — Full LLVM IR output
- `scope_tree.html` — HTML visualization of semantic scopes

Visualize the AST:
```bash
dot -Tpng hello_debug/ast.dot -o ast.png
```

## Command-Line Arguments

Programs can accept command-line arguments by defining `main` with `argc` and `argv` parameters:

```sun
using sun;

function main(argc: i32, argv: ptr<ptr<i8>>) i32 {
    println("Number of arguments received:");
    println_i32(argc);
    return 0;
}

manifest {
    moons: ["stdlib.moon"]
}
```

```bash
sun echo.sun hello world
Number of arguments received:
3
```

### Echo Example

A complete program that echoes all command-line arguments:

```sun
using sun;

function main(argc: i32, argv: ptr<ptr<i8>>) i32 {
    for (var i: i32 = 0; i < argc; i = i + 1) {
        println(argv[i]);
    }
    return 0;
}

manifest {
    moons: ["stdlib.moon"]
}
```

```bash
./echo a b c
./echo
a
b
c
```

## Matrix Operations

### Creating and Indexing Matrices

Using the standard library `Matrix<T>` class:

```sun
using sun;

function main() i32 {
    var allocator = make_heap_allocator();
    
    // Create a 3x3 matrix
    var m = Matrix<i32>(allocator, [3, 3]);
    
    // Use bracket syntax for indexing
    m[0, 0] = 1;  m[0, 1] = 2;  m[0, 2] = 3;
    m[1, 0] = 4;  m[1, 1] = 5;  m[1, 2] = 6;
    m[2, 0] = 7;  m[2, 1] = 8;  m[2, 2] = 9;
    
    // Read elements
    var sum = m[0, 0] + m[1, 1] + m[2, 2];  // Diagonal sum: 15
    
    return sum;
}

manifest {
    moons: ["stdlib.moon"]
}
```

### Multi-dimensional Matrices

## Classes and Constructors

### Basic Class

```sun
using sun;

class Point {
    var x: f64;
    var y: f64;
    
    function init(x_: f64, y_: f64) {
        this.x = x_;
        this.y = y_;
    }
    
    function distance(other: ref Point) f64 {
        var dx = this.x - other.x;
        var dy = this.y - other.y;
        return _sqrt(dx * dx + dy * dy);
    }
}

function main() i32 {
    var p1 = Point(0.0, 0.0);
    var p2 = Point(3.0, 4.0);
    
    var dist = p1.distance(p2);  // 5.0
    println_f64(dist);
    
    return 0;
}

manifest {
    moons: ["stdlib.moon"]
}
```

### Generic Class

```sun
class Pair<T> {
    var first: T;
    var second: T;
    
    function init(a: T, b: T) {
        this.first = a;
        this.second = b;
    }
    
    function swap() void {
        var temp = this.first;
        this.first = this.second;
        this.second = temp;
    }
}

function main() i32 {
    var p = Pair<i32>(10, 20);
    p.swap();
    return p.first;  // 20
}
```

## Interfaces

```sun
using sun;

interface Drawable {
    function draw() void;
}

class Circle implements Drawable {
    var radius: f64;
    
    function init(r: f64) {
        this.radius = r;
    }
    
    function draw() void {
        println("Drawing circle with radius:");
        println_f64(this.radius);
    }
}

manifest {
    moons: ["stdlib.moon"]
}

class Square implements Drawable {
    var side: f64;
    
    function init(s: f64) {
        this.side = s;
    }
    
    function draw() void {
        println("Drawing square with side:");
        println_f64(this.side);
    }
}

function render(shape: ref Drawable) void {
    shape.draw();
}

function main() i32 {
    var c = Circle(5.0);
    var s = Square(10.0);
    
    render(c);  // "Drawing circle with radius: 5.0"
    render(s);  // "Drawing square with side: 10.0"
    
    return 0;
}
```

## Error Handling

```sun
function divide(a: i32, b: i32) i32, IError {
    if (b == 0) {
        throw 1;  // Division by zero error
    }
    return a / b;
}

function main() i32 {
    try {
        var x = divide(10, 2);   // OK: 5
        var y = divide(10, 0);   // Throws
        return x + y;
    } catch (e: IError) {
        println("Error: division by zero");
        return -1;
    }
}
```

## Lambdas and Closures

```sun
function main() i32 {
    var multiplier: i32 = 10;
    
    // Lambda capturing 'multiplier' from environment
    var scale = lambda (x: i32) i32 {
        return x * multiplier;
    };
    
    var result: i32 = 0;
    result = result + scale(1);  // 10
    result = result + scale(2);  // 20
    result = result + scale(3);  // 30
    result = result + scale(4);  // 40
    result = result + scale(5);  // 50
    
    return result;  // 150
}
```

## Memory Management

### Heap Allocation with Unique

```sun
using sun;

class Point {
    var x: f64;
    var y: f64;
    
    function init(px: f64, py: f64) {
        this.x = px;
        this.y = py;
    }
}

function main() i32 {
    var allocator = make_heap_allocator();
    
    // Allocate on heap, wrap in Unique<T> for automatic cleanup
    var p = Unique<Point>(allocator.create<Point>(3.0, 4.0));
    
    println_f64(p.get().x);  // 3.0
    println_f64(p.get().y);  // 4.0
    
    return 0;
    // p.deinit() called automatically, memory freed
}

manifest {
    moons: ["stdlib.moon"]
}
```

### Reference Parameters

```sun
function double_value(x: ref i32) void {
    x = x * 2;  // Modifies the original
}

function main() i32 {
    var value: i32 = 21;
    double_value(value);
    return value;  // 42
}
```
